#@type:grammar;@schema:procedure;
INTERSECTS{
PRECONDITION: START
ACTION:
	SET "RESULT=FALSE";
	IF !A|!B -> CONTINUE;
	IF OVERLAP_EXISTS_PRIMITIVE(A,B) -> SET "RESULT=TRUE";
	CONTINUE;
STATE:{IN=A|B;OUT=RESULT;TAG=GENERAL}
}

#@type:grammar;@schema:procedure;
CLAMP{
PRECONDITION: START
ACTION:
	SET "RESULT=VALUE";
	IF VALUE<MIN -> SET "RESULT=MIN";
	IF VALUE>MAX -> SET "RESULT=MAX";
	CONTINUE;
STATE:{IN=VALUE|MIN|MAX;OUT=RESULT;TAG=GENERAL}
}

#@type:grammar;@schema:procedure;
COALESCE{
PRECONDITION:START
ACTION:
	SET "RESULT=DEFAULT";
	IF VALUE_PRESENT -> SET "RESULT=VALUE";
	CONTINUE;
STATE:{IN=VALUE|VALUE_PRESENT|DEFAULT;OUT=RESULT;TAG=GENERAL}
}
